** Tous les composants qui utilisent du @ModuleDashboardPageAction ou @ModuleDashboardPageGetter : **

*** Injecter le store local du DB (première ligne de la classe du composant) ***
    @Inject('storeNamespace') readonly storeNamespace!: string;

*** Ajouter les méthodes publiques d'accès au store ***

    // Accès dynamiques Vuex
    public vuexGet<T>(getter: string): T {
        return (this.$store.getters as any)[`${this.storeNamespace}/${getter}`];
    }
    public vuexAct<A>(action: string, payload?: A) {
        return this.$store.dispatch(`${this.storeNamespace}/${action}`, payload);
    }
